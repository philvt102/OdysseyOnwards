## 2. The Complete Interactive Script

<script define:vars={{ systems }}>
    // Guard against stringified build variables
    const verifiedSystems = typeof systems === 'string' ? JSON.parse(systems) : systems;

    const dropdown = document.getElementById('dropdown-menu');
    const checkboxes = document.querySelectorAll('#checks input[type="checkbox"]');
    const decadeButtons = document.querySelectorAll('.decade-btn');
    const manuButtons = document.querySelectorAll('.manu-btn');

    let activeDecade = null;
    let activeManu = null;
    let hoverTimeout = null;

    function getCheckedTypes() {
        return Array.from(checkboxes).filter(cb => cb.checked).map(cb => cb.value);
    }

    function parseJsonTypes(typeString) {
        if (!typeString) return [];
        return typeString.replace(/[\[\]]/g, '').split(',').map(t => t.trim());
    }

    function passesTypeCheck(item, checkedTypes) {
        const itemTypes = parseJsonTypes(item.type);
        if (checkedTypes.includes('other')) {
            const knownTypes = ['console', 'addon', 'tabletop', 'micro', 'pong', 'handheld', 'pnp'];
            if (!itemTypes.some(t => knownTypes.includes(t))) return true;
        }
        return itemTypes.some(t => checkedTypes.includes(t));
    }

    // --- 10-Column Decade HTML ---
    function buildDecadeHTML(startYear) {
        let html = '';
        const checkedTypes = getCheckedTypes();

        for (let year = startYear; year < startYear + 10; year++) {
            const matches = verifiedSystems.filter(item => {
                const releaseYear = parseInt(item.releaseYear, 10);
                return releaseYear === year && passesTypeCheck(item, checkedTypes);
            });

            html += `
                <div class="dropdown-col">
                    <h4>${year}</h4>
                    <div class="col-links">
                        ${matches.length > 0 
                            ? matches.map(item => `<a href="#${item.name.replace(/\s+/g, '-')}">${item.name}</a>`).join('')
                            : `<span class="no-systems">—</span>`
                        }
                    </div>
                </div>
            `;
        }
        return html;
    }

    // --- Single-Column Brand HTML ---
    function buildManuHTML(manufacturerName) {
        const checkedTypes = getCheckedTypes();
        const topFiveBrands = ['atari', 'nintendo', 'sega', 'sony', 'microsoft'];

        const matches = verifiedSystems.filter(systemObj => {
            if (!passesTypeCheck(systemObj, checkedTypes)) return false;

            const itemManu = systemObj.manufacturer ? String(systemObj.manufacturer).trim() : "";
            const itemManuLower = itemManu.toLowerCase();

            if (manufacturerName === 'Everything Else') {
                return !topFiveBrands.includes(itemManuLower) && itemManuLower !== "";
            }
            return itemManuLower === manufacturerName.toLowerCase().trim();
        });

        return `
            <div class="dropdown-col" style="border-right: none;">
                <h4>${manufacturerName}</h4>
                <div class="col-links">
                    ${matches.length > 0 
                        ? matches.map(systemObj => `<a href="#${systemObj.name.replace(/\s+/g, '-')}">${systemObj.name}</a>`).join('')
                        : `<span class="no-systems">No matching systems</span>`
                    }
                </div>
            </div>
        `;
    }

    // --- Event Listeners: Decades ---
    decadeButtons.forEach(btn => {
        btn.addEventListener('mouseenter', (e) => {
            clearTimeout(hoverTimeout);
            activeManu = null;
            activeDecade = parseInt(e.target.dataset.decade, 10);
            
            dropdown.style.left = '';
            dropdown.style.width = '';
            dropdown.classList.remove('single-col');
            
            dropdown.innerHTML = buildDecadeHTML(activeDecade);
            dropdown.classList.remove('hidden');
        });

        btn.addEventListener('mouseleave', () => {
            clearTimeout(hoverTimeout);
            hoverTimeout = setTimeout(() => dropdown.classList.add('hidden'), 200);
        });
    });

    // --- Event Listeners: Manufacturers ---
    manuButtons.forEach(btn => {
        btn.addEventListener('mouseenter', (e) => {
            clearTimeout(hoverTimeout);
            activeDecade = null;
            activeManu = e.target.dataset.manu;

            dropdown.classList.add('single-col');
            dropdown.innerHTML = buildManuHTML(activeManu);
            
            const parentElement = document.getElementById('buttons');
            const parentRect = parentElement.getBoundingClientRect();
            const btnRect = e.target.getBoundingClientRect();
            
            const offsetLeft = btnRect.left - parentRect.left;
            dropdown.style.left = `${offsetLeft}px`;
            dropdown.style.width = '250px';
            
            dropdown.classList.remove('hidden');
        });

        btn.addEventListener('mouseleave', () => {
            clearTimeout(hoverTimeout);
            hoverTimeout = setTimeout(() => dropdown.classList.add('hidden'), 200);
        });
    });

    // --- Dropdown Shell Hover Bridges ---
    dropdown.addEventListener('mouseenter', () => clearTimeout(hoverTimeout));
    dropdown.addEventListener('mouseleave', () => {
        clearTimeout(hoverTimeout);
        hoverTimeout = setTimeout(() => dropdown.classList.add('hidden'), 200);
    });

    // --- Checkbox Realtime Redraws ---
    checkboxes.forEach(cb => {
        cb.addEventListener('change', () => {
            if (dropdown.classList.contains('hidden')) return;
            if (activeDecade) dropdown.innerHTML = buildDecadeHTML(activeDecade);
            if (activeManu) dropdown.innerHTML = buildManuHTML(activeManu);
        });
    });</script>

// --- Inverted Toggle Engine: Default Off ---const layoutToggle = document.getElementById('layout-toggle-switch');
function forceLayoutRedraw() {
    if (dropdown.classList.contains('hidden')) return;
    if (activeDecade) dropdown.innerHTML = buildDecadeHTML(activeDecade);
    if (activeManu) dropdown.innerHTML = buildManuHTML(activeManu);
}

layoutToggle.addEventListener('change', (e) => {
    const isSelectAll = e.target.checked;

    checkboxes.forEach(cb => {
        if (isSelectAll) {
            cb.checked = true;
        } else {
            cb.checked = (cb.value === 'console');
        }
    });

    forceLayoutRedraw();
});
// Dynamic state watcher
checkboxes.forEach(cb => {
    cb.addEventListener('change', () => {
        const allChecked = Array.from(checkboxes).every(item => item.checked);
        
        if (allChecked && !layoutToggle.checked) {
            layoutToggle.checked = true;
        } else if (!allChecked && layoutToggle.checked) {
            layoutToggle.checked = false;
        }
    });
});
